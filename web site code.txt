  <script>
    const DATA_PATH = 'data.json';     
    const POLL_INTERVAL_MS = 5000;      
    document.getElementById('pollInt').textContent = (POLL_INTERVAL_MS/1000) + 's';

    function safeText(v){ return (v===undefined || v===null) ? '--' : v; }

    async function fetchAndUpdate(){
      try{
        const resp = await fetch(DATA_PATH + '?_=' + Date.now(), {cache: 'no-store'});
        if(!resp.ok) throw new Error('File not found or server error');
        const data = await resp.json();

        
        const t1 = safeText(data.temperature1);
        const h1 = safeText(data.humidity1);
        const t2 = safeText(data.temperature2);
        const h2 = safeText(data.humidity2);
        const pressure = safeText(data.pressure);
        const mq = safeText(data.airQuality ?? data.mq135 ?? data.mq);
        const gx = safeText(data.gx);
        const gy = safeText(data.gy);
        const gz = safeText(data.gz);
        const ts = safeText(data.timestamp ?? new Date().toLocaleString());

     
        document.getElementById('t1').textContent = t1 + ' °C';
        document.getElementById('h1').textContent = h1 + ' %';
        document.getElementById('t2').textContent = t2 + ' °C';
        document.getElementById('h2').textContent = h2 + ' %';
        document.getElementById('pressure').textContent = pressure + ' hPa';
        document.getElementById('mq').textContent = mq;
        document.getElementById('gx').textContent = gx;
        document.getElementById('gy').textContent = gy;
        document.getElementById('gz').textContent = gz;
        document.getElementById('raw').textContent = JSON.stringify(data);

        document.getElementById('lastUpdate').textContent = 'Последно читање: ' + ts;
        document.getElementById('tsSmall').textContent = 'Последен запис: ' + ts;
        document.getElementById('status').textContent = 'OK';

       
        const dot = document.getElementById('mqDot');
        const mqVal = parseFloat(mq);
        if (!isNaN(mqVal)) {
          if (mqVal < 100) { dot.style.background = 'var(--ok)'; document.getElementById('status').classList.remove('pulse'); }
          else if (mqVal < 300) { dot.style.background = 'var(--warn)'; document.getElementById('status').classList.add('pulse'); }
          else { dot.style.background = 'var(--danger)'; document.getElementById('status').classList.add('pulse'); }
        } else {
          dot.style.background = 'var(--muted)';
        }

      
        const els = document.querySelectorAll('.fade');
        els.forEach(el => { el.style.transform = 'translateX(-6px)'; el.style.opacity = '0.6';
          setTimeout(()=>{ el.style.transform='translateX(0)'; el.style.opacity='1'; }, 200);
        });

      } catch (err) {
        console.error(err);
        document.getElementById('status').textContent = 'No data';
        document.getElementById('tsSmall').textContent = '';
      }
    }

  
    fetchAndUpdate();
    setInterval(fetchAndUpdate, POLL_INTERVAL_MS);
  </script>
